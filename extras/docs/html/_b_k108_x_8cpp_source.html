21  * @ingroup GA03
22  * @brief Gets a givens current register content of the device
23  * @see shadowRegisters;
24  * @param device register address
25  */
26 uint16_t BK108X::getRegister(uint8_t reg)
27 {
28  word16_to_bytes result;
29 
30  Wire.beginTransmission(this->deviceAddress);
31  Wire.write(reg);
32  Wire.endTransmission(false);
33  // delayMicroseconds(1000);
34  Wire.requestFrom(this->deviceAddress, 2);
35  result.refined.highByte = Wire.read();
36  result.refined.lowByte = Wire.read();
37  Wire.endTransmission(true);
38  // delayMicroseconds(1000);
39 
40  shadowRegisters[reg] = result.raw; // Syncs with the shadowRegisters
41 
42  return result.raw;
43 }

46 /**
47  * @ingroup GA03
48  * @brief Sets a given value to the device registers
49  * @details For write operations, the device acknowledge is followed by an eight bit data word latched internally on rising edges of SCLK. The device acknowledges each byte of data written by driving SDIO low after the next falling SCLK edge, for 1 cycle.
50  * @details An internal address counter automatically increments to allow continuous data byte writes, starting with the upper byte of register 02h, followed by the lower byte of register 02h, and onward until the lower byte of the last register is reached. The internal address counter then automatically wraps around to the upper byte of register 00h and proceeds from there until continuous writes end.
51  * @details The registers from 0x2 to 0x07 are used to setup the device. This method writes the array shadowRegisters, elements 8 to 14 (corresponding the registers 0x2 to 0x7 respectively) into the device. See Device registers map in BK108X.h file.
52  * @details To implement this, a register maping was created to deal with each register structure. For each type of register, there is a reference to the array element.
53  *
54  * @see shadowRegisters;
55  *
56  * @param device register address
57  */
58 void BK108X::setRegister(uint8_t reg, uint16_t value)
59 {
60  word16_to_bytes param;
61 
62  param.raw = value;
63 
64  Wire.beginTransmission(this->deviceAddress);
65  Wire.write(reg);
66 
67  Wire.write(param.refined.highByte);
68  Wire.write(param.refined.lowByte);
69  Wire.endTransmission();
70 
71  shadowRegisters[reg] = value; // Syncs with the shadowRegisters
72 
73  // delayMicroseconds(6000);
74 }
